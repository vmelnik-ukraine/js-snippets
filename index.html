Forms changes spectator

Simple script to make forms snapshots and then compare them.

/**
 * Lets create spectator for specific form with id 'form-id'.
 * Also we can just provide selector for area with multiple forms
 */
var formSpectator = new FormSpectator('#form-id');

// We don't change anything, so form is in the same state as before
formSpectator.wasChanges(); // false

// Change some value in form
$('#form-field').val('new value');

// Check form state again
formSpectator.wasChanges(); // true

/**
 * If are you using ajax, you need to reinit form spectator
 * after form replacement
 */
formSpectator.reinit();

More complex example

/**
 * If were changes in form, then we allow to send it to server. 
 * After ajax form replacement we must reinit form spectator
 * so new form values become a new default state for this form
 */
var formSpectator = new FormSpectator('#form-id');

$('#form-submit-btn').click(function() {
    if(formSpectator.wasChanges()) {
        /**
         * Send form by ajax to server and then replace it with form
         * from response...
         */
        formSpectator.reinit();
    } else {
        alert('Nothing to update');
    }
});
